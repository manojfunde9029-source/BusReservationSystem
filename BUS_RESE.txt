//Program for booking seat
void seat()
{
  int n,s;
  printf("Enter the bus number(1-4): ");
  scanf("%d",&n);
  if(n<1||n>4)
  {
    printf("Invalid choice");
  }
  else
  {
  printf("Enter the seat number(1-20): ");
  scanf("%d",&s);
  if(s<1||s>20)
  {
    printf("Invalid seat number");
  }
  else if(bus[n-1][s-1]==1)
	{
	  printf("Seat is already booked");
	}
	else
	{
	  bus[n-1][s-1]=1;
	  printf("Seat is booked succesfully");
	}
  }
}
//Program for cancelling seats
void cancel()
{
  int n,s;
  printf("Enter the bus number(1-4): ");
  scanf("%d",&n);
  if(n<1||n>4)
  {
    printf("Invalid choice");
  }
  else
  {
  printf("Enter the seat number(1-20): ");
  scanf("%d",&s);
  if(s<=1||s>=20)
  {
    printf("Invalid seat number");
  }
  else if(bus[n-1][s-1]==0)
	{
	  printf("Seat is already empty");
	}
	else
	{
	  bus[n-1][s-1]=0;
	  printf("Seat is cancelled succesfully");
	}

  }
}
